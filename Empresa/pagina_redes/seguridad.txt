Para hacer que la pagina sea segura utilice el "validate" junto con el "trim-striplashes-htmlspecialchars" que sirve para limpiar los espacios en blanco las comillas o o barras invertidas,y tambien convierte los caracteres especiales en entidades HTML


    function validate($data){
        $data = trim($data);
        $data = stripslashes($data);
        $data = htmlspecialchars($data);
        return $data;
    }

    $usuario = validate($_POST['usuario']); 
    $contraseña = validate($_POST['contraseña']);
